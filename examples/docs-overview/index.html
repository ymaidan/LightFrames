🚀 Quick Start Guide

1. Create HTML Structure

<!DOCTYPE html>
<html>
<head>
  <title>My LightFrame App</title>
</head>
<body>
  <div id="app"></div>
  
  <script src="./src/core.js"></script>
  <script src="./src/events.js"></script>
  <script src="./src/state.js"></script>
  <script src="./src/framework.js"></script>
  <script src="./app.js"></script>
</body>
</html>

2. Create Your First Component

// app.js
const { DOM, Store } = LightFrame;

// Create a store for state management
const store = new Store({ count: 0 });

// Create a render function
function render() {
  const state = store.getState();
  
  const app = DOM.createElement('div', { class: 'app' }, [
    DOM.createElement('h1', {}, ['My LightFrame App']),
    DOM.createElement('p', {}, [`Count: ${state.count}`]),
    DOM.createElement('button', {
      onclick: () => store.setState({ count: state.count + 1 })
    }, ['Increment'])
  ]);
  
  DOM.render(app, document.getElementById('app'));
}

// Subscribe to state changes
store.subscribe(render);

// Initial render
render();

3. Run Your App

npm start

Open http://localhost:3000 in your browser!
